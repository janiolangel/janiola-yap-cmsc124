package cmsc124_lab1

// Enum for all possible token types
enum class TokenType {
    // Single-character tokens
    LEFT_PAREN, RIGHT_PAREN,
    LEFT_BRACE, RIGHT_BRACE,
    COMMA, DOT, MINUS, PLUS,
    SEMICOLON, SLASH, STAR,
    EQUAL, LESS, GREATER, BANG,

    // Multi-character operators
    EQUAL_EQUAL,
    BANG_EQUAL,
    LESS_EQUAL,
    GREATER_EQUAL,

    // Literals
    STRING,
    NUMBER,

    EOF
}

// cmsc124_lab1.Token class
data class Token(
    val type: TokenType,
    val lexeme: String,
    val literal: Any?,
    val line: Int
)
